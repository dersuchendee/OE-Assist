SELECT ?person ?union ?role ?roleTitle ?roleStartDate ?roleEndDate
WHERE {

  ?person :isMemberOf ?membership .
  ?membership :membershipInUnion ?union .
  ?membership :hasUnionRole ?role .
  ?role :unionRoleTitle ?roleTitle .

}
