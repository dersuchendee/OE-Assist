PREFIX : <http://www.example.org/test#>
PREFIX rdfs: <http://www.w3.org/2000/01/rdf-schema#>

SELECT ?majorClass ?majorClassLabel ?subClass ?subClassLabel

  OPTIONAL { ?subClass rdfs:label ?subClassLabel }
}
