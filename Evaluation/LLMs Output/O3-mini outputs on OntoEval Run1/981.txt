PREFIX : <http://www.example.org/test#>

SELECT ?book ?bookLabel ?numberOfPages
WHERE {
  ?book a :Cl_Book .
  ?book :hasDetails ?details .
  ?details :numberOfPages ?numberOfPages .
  OPTIONAL { ?book rdfs:label ?bookLabel. }
}
