PREFIX : <http://example.org/ontology#>
PREFIX xsd: <http://www.w3.org/2001/XMLSchema#>

SELECT ?calendar ?event ?start ?end
WHERE {
  ?calendar a :Calendar ;
            :containsEvent ?event .
  ?event :hasTimePeriod ?timePeriod .
  ?timePeriod :startTime ?start ;
              :endTime ?end .

           ?end <= "2023-04-01T23:59:59Z"^^xsd:dateTime )
