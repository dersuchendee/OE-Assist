PREFIX : <http://www.example.org/test#>
PREFIX rdfs: <http://www.w3.org/2000/01/rdf-schema#>

WHERE {
  ?book a :Cl_Book ;
        :pageCount ?pageCount .
  OPTIONAL { ?book rdfs:label ?bookLabel . }
