--------------------------------------------------

PREFIX rdf: <http://www.w3.org/1999/02/22-rdf-syntax-ns#>

WHERE {
  ?production a :Cl_Production .
  ?production :hasPremier ?premier .
}

HAVING(COUNT(?premier)=1)
