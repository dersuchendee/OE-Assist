PREFIX : <http://example.org/ontology#>
PREFIX rdfs: <http://www.w3.org/2000/01/rdf-schema#>

WHERE {
  ?hospital a :Hospital .
  OPTIONAL { ?hospital rdfs:label ?hospitalLabel. }

  OPTIONAL { ?city rdfs:label ?cityLabel. }
}
--------------------------------
